// 1. БАЗА ДАННЫХ ПЕРСОНАЖЕЙ (МАГАЗИН) const CHARACTER_SHOP = [ { id: "default", emoji: "😊", name: "Улыбака", price: 0, purchased: true, speed: 5, health: 100 }, { id: "demon", emoji: "👿", name: "Демон", price: 100, purchased: false, speed: 6, health: 120 }, { id: "soldier", emoji: "🪖", name: "Ветеран", price: 250, purchased: false, speed: 4, health: 200 }, { id: "bomb", emoji: "💣", name: "Подрывник", price: 400, purchased: false, speed: 5, health: 80 }, { id: "assassin", emoji: "🗡", name: "Ассасин", price: 500, purchased: false, speed: 8, health: 90 } ]; // 2. СОСТОЯНИЕ ИГРОКА (СОХРАНЕНИЕ) let playerState = { coins: 150, // Текущие монеты игрока selectedCharacter: "default" // Экипированный персонаж }; // 3. ФУНКЦИЯ ПОКУПКИ И СМЕНЫ ПЕРСОНАЖА function buyOrSelectCharacter(characterId) { const char = CHARACTER_SHOP.find(c => c.id === characterId); if (!char) return "Персонаж не найден"; // Если уже куплен — просто экипируем if (char.purchased) { playerState.selectedCharacter = characterId; return `Экипирован: ${char.emoji} ${char.name}`; } // Если не куплен — проверяем баланс монет if (playerState.coins >= char.price) { playerState.coins -= char.price; char.purchased = true; playerState.selectedCharacter = characterId; return `Успешно куплен и экипирован: ${char.emoji} ${char.name}`; } else { return `Недостаточно монет! Нужно еще ${char.price - playerState.coins} 💰`; } } // 4. ФУНКЦИЯ ОТРИСОВКИ ИГРОКА НА ЭКРАНЕ (Canvas) function drawPlayer(ctx, playerX, playerY) { // Находим текущего выбранного персонажа const currentChar = CHARACTER_SHOP.find(c => c.id === playerState.selectedCharacter); ctx.font = "40px Arial"; // Размер смайлика ctx.textAlign = "center"; ctx.textBaseline = "middle"; // Рисуем emoji вместо обычного круга (шарика) ctx.fillText(currentChar.emoji, playerX, playerY); }
